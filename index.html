// 제품 그룹 정의 ← 여기서 수정하세요 var PRODUCT_GROUPS = { '메인제품': ['헤드레스트', '핸들커버', '방석', '킥매트', '허리쿠션', '프리백커버'], '기타제품': ['대시보드커버', '데스크매트', '라나매트', '루미락', '모네틱', '사운드픽셀', '심포라', '전자기기', '하이퍼젯'], '공통': ['공통'] }; var PRODUCT_GROUP_COLORS = {'메인제품': '#4f8ef7', '기타제품': '#1D9E75', '공통': '#888780'}; // 제품 탭 동적 생성 function buildProductTabs(rows) { var products = {}; for (var i = 0; i < rows.length; i++) { var p = (rows[i]['제품'] || '').trim(); if (p) products[p] = true; } var allProducts = Object.keys(products).sort(); var html = ''; var groupNames = Object.keys(PRODUCT_GROUPS); for (var gi = 0; gi < groupNames.length; gi++) { var groupName = groupNames[gi]; var groupColor = PRODUCT_GROUP_COLORS[groupName] || '#888'; var groupProds = PRODUCT_GROUPS[groupName].filter(function(p){ return products[p]; }); if (groupProds.length === 0) continue; html += '' + groupName + ''; html += ''; for (var pi = 0; pi < groupProds.length; pi++) { html += ''; } } var ungrouped = allProducts.filter(function(p) { for (var gi = 0; gi < groupNames.length; gi++) { if (PRODUCT_GROUPS[groupNames[gi]].indexOf(p) !== -1) return false; } return true; }); if (ungrouped.length > 0) { html += '미분류'; for (var ui = 0; ui < ungrouped.length; ui++) { html += ''; } } document.getElementById('productTabs').innerHTML = html; var tabs = document.querySelectorAll('#productTabs .tab'); for (var i = 0; i < tabs.length; i++) { var dp = tabs[i].getAttribute('data-product'); tabs[i].classList.toggle('active', dp === S.product || dp === '__group__' + S.productGroup); } } function setProductGroup(btn, groupName) { var tabs = document.querySelectorAll('#productTabs .tab'); for (var i = 0; i < tabs.length; i++) tabs[i].classList.remove('active'); btn.classList.add('active'); S.product = 'all'; S.productGroup = groupName; campaignCache = {}; sheetsCache.aimban && (sheetsCache.aimban._filteredNSA = null); renderWithData(); }
